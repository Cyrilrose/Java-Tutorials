class ChangeMaker

import java.util.Scanner;
 public class ChangeMaker
 {
     public static void main (String [] args)
     {
         int amount, originalAmount,
             quarters, dimes, nickels, pennies;
         System.out.println ("Enter a whole number from 1 to 99.");
         System.out.println ("I will find a combination of coins");
         System.out.println ("that equals that amount of change.");
         Scanner keyboard = new Scanner (System.in);
         amount = keyboard.nextInt ();
         originalAmount = amount;
         quarters = amount / 25;
         amount = amount % 25;
         dimes = amount / 10;
         amount = amount % 10;
         nickels = amount / 5;
         amount = amount % 5;
         pennies = amount;
         System.out.println (originalAmount +
                 " cents in coins can be given as:");
         System.out.println (quarters + " quarters");
         System.out.println (dimes + " dimes");
         System.out.println (nickels + " nickels and");
         System.out.println (pennies + " pennies");
     }
 }

Scanner Example

class ScannerDemo

import java.util.Scanner;
public class ScannerDemo
{
    public static void main (String [] args)
    {
        Scanner keyboard = new Scanner (System.in);
        System.out.println ("Enter two whole numbers");
        System.out.println ("separated by one or more spaces:");
        int n1, n2;
        n1 = keyboard.nextInt ();
        n2 = keyboard.nextInt ();
        System.out.println ("You entered " + n1 + " and " + n2);
        System.out.println ("Next enter two numbers.");
        System.out.println ("A decimal point is OK.");
        double d1, d2;
        d1 = keyboard.nextDouble ();
        d2 = keyboard.nextDouble ();
        System.out.println ("You entered " + d1 + " and " + d2);
        System.out.println ("Next enter two words:");
        String s1, s2;
        s1 = keyboard.next ();
        s2 = keyboard.next ();
        System.out.println ("You entered \"" +
                s1 + "\" and \"" + s2 + "\"");
        s1 = keyboard.nextLine (); //To get rid of '\n'
        System.out.println ("Next enter a line of text:");
        s1 = keyboard.nextLine ();
        System.out.println ("You entered: \"" + s1 + "\"");
    }
}

	  //numberOfBaskets assigns the variable equal to 10 
	  //eggsPerBasket assigns the variable equal to 6  
			 
	  totalEggs = numberOfBaskets * eggsPerBasket; 
	  //This line does two things: 
	  //Multiply the value of numberOfBaskets and eggsPerBasket first. 
	  //the product of two variables get assigned to totalEggs 
			 
	  System.out.println (eggsPerBasket + " eggs per basket and"); 
	  //It prints a longer string by combining 2 different types of variables 
	  //an integer: eggsPerBasket(6) and a string: " eggs per basket and" 
	  //the "+" operator does the job of concatenator

	  originalAmount = amount;
	  //Assign amount entered from user to originalAmount
			
	  quarters = amount / 25;
	  //gets the maximum quarters from user entered amount
	  //and assigned to parameter quarters
			
	  amount = amount % 25;
	  //gets the remainder of amount after take away quarters
	  //and reassigned to parameter amount
			
	  dimes = amount / 10;
	  //gets the maximum dimes from leftover amount
	  //and assigned to parameter dimes
			
	  amount = amount % 10;
	  //gets the remainder of amount after take away dimes
	  //and reassigned to parameter amount
			
	  nickels = amount / 5;
	  //gets the maximum nickels from leftover amount
	  //and assigned to parameter nickels
			
	  amount = amount % 5;
	  //gets the remainder of amount after take away nickels
	  //and reassigned to parameter amount
