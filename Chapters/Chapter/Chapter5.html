Person Class Example

class Person

public class Person
{
  private String name;
  private int age;
  public Person()
  {
    this.name = "no name";
    this.age = 0;
  }
  public Person(String initialName, int initialAge)
  {
    this.name = initialName;
    this.age = initialAge;
  }
  public void setAll(String newName, int newAge)
  {
    this.name = newName;
    this.age = newAge;
  }
  public String getName()
  {
    return this.name;
  }
  public int getAge()
  {
    return this.age;
  }
  public boolean hasSameName(Person otherPerson)
  {
    return this.name.equalsIgnoreCase(otherPerson.name);
  }
}

Deep look of class Person

  private String name;
  //declare a String type variable name in private
  /*note private can't get direct access from outside of this class*/
  
  /*default constractor for Person class*/
  public Person()
  {
    this.name = "no name";
  //this.name means the name parameter of this Person class
    this.age = 0;
  }
  
  /*constractor of Person class with 2 parameters for initialization*/
  public Person(String initialName, int initialAge)
  {
    this.name = initialName;
	//sets name to parameter initialName
    this.age = initialAge;
	sets age to parameter initialAge 
  }
  
  /*void type of setAll method with 2 parameters*/
  /*note the void type does not return anything*/
  /*note public means this method is accessible from outside of this class*/
  public void setAll(String newName, int newAge)
  {
  /*sets name and age for Person class*/
    this.name = newName;
    this.age = newAge;
  }

Deep look of class Person(Cont.)

  /*String type of getName method*/
  /*note exctpt void type, any other types need to return that type of value*/
  public String getName()
  {
    return this.name;
  //return the String type parameter name
  }
  /*a boolean type method hasSameName with a Person type object otherPerson*/
  public boolean hasSameName(Person otherPerson)
  {
    return this.name.equalsIgnoreCase(otherPerson.name);
  //this line means the current Person object calls the name parameter
  //then the name parameter calls equalsIgnoreCase method from String class  
  //to compare with another Person class, which also calls the name parameter
  }
